```Java
class Solution {
    public int maxAncestorDiff(TreeNode root) {
        int[] maxDiff = new int[1];
        maxDiff[0] = Integer.MIN_VALUE;
        helper(root, maxDiff);
        return maxDiff[0];
    }
    
    private int[] helper(TreeNode root, int[] maxDiff) {
        if (root == null) {
            return new int[] {0, 0};
        } else if (root.left == null && root.right == null) {
            return new int[] {root.val, root.val};
        }
        // return the minimum and maximum value in its subtree
        int[] left = helper(root.left, maxDiff);
        int[] right = helper(root.right, maxDiff);
        // determine the minimum and maximum value in current root's subtree
        int min = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;
        if (root.left != null) {
            // root has a left child, we can update the maxDiff between the nodes in
            // its left subtree and itself
            maxDiff[0] = Math.max(maxDiff[0], 
                     Math.max(Math.abs(root.val - left[0]), Math.abs(root.val - left[1])));
            min = Math.min(left[0], root.val);
            max = Math.max(left[1], root.val);
        }
        if (root.right != null) {
            // root has a right child, we can update the maxDiff between the nodes in
            // its right subtree and itself
            maxDiff[0] = Math.max(maxDiff[0], 
                     Math.max(Math.abs(root.val - right[0]), Math.abs(root.val - right[1])));
            min = Math.min(min, Math.min(right[0], root.val));
            max = Math.max(max, Math.max(right[1], root.val));
        }
        return new int[] {min, max};
    }
}
```
